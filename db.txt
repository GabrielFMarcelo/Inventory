CREATE TABLE printing_types (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(50) NOT NULL UNIQUE,   -- e.g., 'CTP', 'Digital', 'Riso'
    base_cost DECIMAL(10,2) DEFAULT 0, -- fixed amount added (like 300 for CTP, 15 for Riso)
    per_sheet_cost DECIMAL(10,4) DEFAULT 0, -- per sheet per layer
    apply_to_paper_cost TINYINT(1) DEFAULT 0, -- if it modifies paper cost (like Riso +15)
    effective_date DATE DEFAULT CURRENT_DATE
);
